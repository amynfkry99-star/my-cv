نظام إدارة المطعم بلغة جافا

كود نظام إدارة المطعم باستخدام لغة جافا وواجهة Swing

RestaurantManagementSystem.java

import javax.swing.*;
import java.awt.*;
import java.awt.event.*;
import java.sql.*;
import java.util.ArrayList;

public class RestaurantManagementSystem extends JFrame {
    
    // عناصر الواجهة
    private JTabbedPane tabbedPane;
    private JPanel menuPanel, ordersPanel, inventoryPanel;
    
    // قوائم البيانات
    private ArrayList<MenuItem> menuItems;
    private ArrayList<Order> orders;
    
    public RestaurantManagementSystem() {
        super("نظام إدارة المطعم");
        initialize();
        loadData();
        createUI();
    }
    
    private void initialize() {
        menuItems = new ArrayList<>();
        orders = new ArrayList<>();
        
        // تعيين إعدادات النافذة
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setSize(1000, 700);
        setLocationRelativeTo(null);
    }
    
    private void loadData() {
        // تحميل البيانات من قاعدة البيانات
        try {
            Connection conn = DriverManager.getConnection("jdbc:sqlite:restaurant.db");
            Statement stmt = conn.createStatement();
            
            // تحميل قائمة الطعام
            ResultSet rs = stmt.executeQuery("SELECT * FROM menu_items");
            while (rs.next()) {
                MenuItem item = new MenuItem(
                    rs.getInt("id"),
                    rs.getString("name"),
                    rs.getString("category"),
                    rs.getDouble("price"),
                    rs.getInt("stock")
                );
                menuItems.add(item);
            }
            
            conn.close();
        } catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "خطأ في تحميل البيانات: " + e.getMessage());
        }
    }
    
    private void createUI() {
        tabbedPane = new JTabbedPane();
        
        // إنشاء لوحات التبويب
        createMenuPanel();
        createOrdersPanel();
        createInventoryPanel();
        
        add(tabbedPane);
    }
    
    private void createMenuPanel() {
        menuPanel = new JPanel(new BorderLayout());
        
        // إنشاء جدول قائمة الطعام
        String[] columns = {"ID", "اسم الصنف", "الفئة", "السعر", "المخزون"};
        DefaultTableModel model = new DefaultTableModel(columns, 0);
        
        for (MenuItem item : menuItems) {
            model.addRow(new Object[]{
                item.getId(),
                item.getName(),
                item.getCategory(),
                item.getPrice(),
                item.getStock()
            });
        }
        
        JTable menuTable = new JTable(model);
        JScrollPane scrollPane = new JScrollPane(menuTable);
        
        menuPanel.add(scrollPane, BorderLayout.CENTER);
        
        // إضافة أزرار التحكم
        JPanel buttonPanel = new JPanel();
        JButton addButton = new JButton("إضافة صنف");
        JButton editButton = new JButton("تعديل");
        JButton deleteButton = new JButton("حذف");
        
        buttonPanel.add(addButton);
        buttonPanel.add(editButton);
        buttonPanel.add(deleteButton);
        menuPanel.add(buttonPanel, BorderLayout.SOUTH);
        
        tabbedPane.addTab("قائمة الطعام", menuPanel);
    }
    
    // باقي دوال النظام...
    
    public static void main(String[] args) {
        SwingUtilities.invokeLater(() -> {
            new RestaurantManagementSystem().setVisible(true);
        });
    }
}
